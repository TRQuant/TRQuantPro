        æ‰§è¡Œå•ä¸ªæ­¥éª¤

            # è·å–æ­¥éª¤é…ç½®
            step = self.dependency_manager.step_map.get(step_id)

                raise ValueError(f"æœªçŸ¥æ­¥éª¤: {step_id}")
            
            # è·å–æ‰§è¡Œæ–¹æ³•
            method_name = step.orchestrator_method
            if not method_name:
                raise ValueError(f"æ­¥éª¤ {step_id} æœªé…ç½®æ‰§è¡Œæ–¹æ³•")
            
            method = getattr(self.orchestrator, method_name)
            if not method:
                raise ValueError(f"æ–¹æ³• {method_name} ä¸å­˜åœ¨")
            
            # æ‰§è¡Œæ­¥éª¤
            self.step_progress.emit(step_id, 10, "å¼€å§‹æ‰§è¡Œ...")

                step_id,
                StepStatus.COMPLETED,
                result={

                    'summary': result.summary,
                    'details': result.details

                progress=100
            )

                'success': result.success,
                'summary': result.summary,
                'details': result.details
            })
            
        except Exception as e:
            logger.error(f"æ­¥éª¤ {step_id} æ‰§è¡Œå¤±è´¥: {e}", exc_info=True)

                step_id,
                StepStatus.FAILED,
                error=str(e),
                progress=0

            self.step_failed.emit(step_id, str(e))
    
    def execute_full_workflow(self):
        """æ‰§è¡Œå®Œæ•´å·¥ä½œæµ"""

            # è·å–æ‰§è¡Œé¡ºåº
            execution_order = self.dependency_manager.get_execution_order()
            
            # æ‰§è¡Œæ¯ä¸ªæ­¥éª¤

            # ç”Ÿæˆå®Œæ•´å·¥ä½œæµç»“æœ
            completed_steps = self.state_manager.get_completed_steps()
            failed_steps = self.state_manager.get_failed_steps()
            
            full_result = {
                'success': len(failed_steps) == 0,
                'completed_steps': completed_steps,
                'failed_steps': failed_steps,
                'total_steps': len(execution_order)
            }
            
            self.workflow_completed.emit(full_result)
            
        except Exception as e:
            logger.error(f"å®Œæ•´å·¥ä½œæµæ‰§è¡Œå¤±è´¥: {e}", exc_info=True)
